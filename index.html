Annotation with Roboflow

Every image was annotated using Roboflow's web UI — bounding boxes drawn precisely around each pen, labeled with its class.

STEP 01

Class Annotation

1 0.501 0.500 0.372 0.130 # capped

Dataset Versioning

STEP 04

Export & Download

Labels downloaded into final_dataset/labels/ with matching image names in final_dataset/images/. The training script pairs these automatically.

Implementation

Two core scripts power this system. train.py prepares data and trains the model; classify.py runs inference.

train.py — Data curation & YOLOv8 training
import os
import shutil
import argparse
from pathlib import Path
from sklearn.model_selection import train_test_split
from ultralytics import YOLO

# --- Configuration ---
# Based on your input: uncapped = 0, capped = 1
SOURCE_DIR = "final_dataset"
YOLO_DIR = "yolo_dataset"

def setup_yolo_dataset():
    """
    Organizes the final_dataset into YOLO train/val format.
    Pairs every .jpg/.jpeg/.png image with its matching .txt label file.
    Performs an 80/20 stratified split for reproducibility.
    """
    print("Organizing dataset for YOLO...")

    if os.path.exists(YOLO_DIR):
        shutil.rmtree(YOLO_DIR)

    for split in ['train', 'val']:
        os.makedirs(f"{YOLO_DIR}/{split}/images", exist_ok=True)
        os.makedirs(f"{YOLO_DIR}/{split}/labels", exist_ok=True)

    img_dir = Path(f"{SOURCE_DIR}/images")
    lbl_dir = Path(f"{SOURCE_DIR}/labels")

    all_images = list(img_dir.glob("*.*"))
    data_pairs = []

    for img_path in all_images:
        if img_path.suffix.lower() in ['.jpg', '.jpeg', '.png']:
            lbl_path = lbl_dir / f"{img_path.stem}.txt"
            if lbl_path.exists():
                data_pairs.append((img_path, lbl_path))

    if not data_pairs:
        raise ValueError(f"No matching images and labels found in {SOURCE_DIR}!")

    # 80% train, 20% val — random_state=42 ensures reproducibility
    train_data, val_data = train_test_split(
        data_pairs, test_size=0.2, random_state=42
    )

    def move_files(data_list, split_name):
        for img_path, lbl_path in data_list:
            shutil.copy(img_path, f"{YOLO_DIR}/{split_name}/images/{img_path.name}")
            shutil.copy(lbl_path, f"{YOLO_DIR}/{split_name}/labels/{lbl_path.name}")

    move_files(train_data, 'train')
    move_files(val_data, 'val')
    print(f"Dataset ready! Train: {len(train_data)} | Val: {len(val_data)}")


def create_yaml():
    """
    Creates the YAML config file that YOLO uses to locate the dataset
    and understand the class mapping (0=uncapped, 1=capped).
    """
    yaml_content = f"""
path: {os.path.abspath(YOLO_DIR)}
train: train/images
val: val/images

names:
  0: uncapped
  1: capped
"""
    with open("pen_data.yaml", "w") as f:
        f.write(yaml_content.strip())
    print("Created pen_data.yaml")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument('--device', type=str, default='cpu')
    parser.add_argument('--epochs', type=int, default=100)
    args = parser.parse_args()

    setup_yolo_dataset()
    create_yaml()

    print("\n--- Initializing YOLOv8 Nano ---")
    model = YOLO("yolov8n.pt")  # Nano variant: fastest, lightest

    results = model.train(
        data="pen_data.yaml",
        epochs=args.epochs,
        imgsz=640,
        batch=16,
        device='cpu' if args.device == 'cpu' else 0,
        name="pen_cap_model",
        patience=25,          # Stop early if no improvement for 25 epochs
        lr0=0.01,             # Initial learning rate
        augment=True,         # Random flips, crops, color jitter, etc.
        label_smoothing=0.1,  # Soft targets reduce overconfidence
        cls=1.5,              # Higher weight on classification loss
        exist_ok=True
    )
classify.py — Inference & detection
import os
os.environ["CUDA_VISIBLE_DEVICES"] = "-1"   # Force CPU inference

import sys
import argparse
from ultralytics import YOLO


def detect_pen_cap(image_path, show_image=True):
    """
    Loads the best trained weights and runs inference on a single image.

    A deliberately low confidence threshold (0.15) is used to minimize
    false negatives — in production, missing a cap (false negative) is
    far more costly than a false alarm (false positive).
    """
    model_path = "runs/detect/pen_cap_model2/weights/best.pt"

    if not os.path.exists(model_path):
        print(f"\n[ERROR]: Could not find the trained model at {model_path}.")
        return

    try:
        model = YOLO(model_path)
    except Exception as e:
        print(f"Error loading model: {e}")
        return

    print(f"\nAnalyzing image: {image_path}...")

    # conf=0.15 → very sensitive; catches borderline uncapped cases
    # iou=0.45  → standard overlap threshold for non-max suppression
    results = model(image_path, conf=0.15, iou=0.45)

    for result in results:
        boxes = result.boxes

        if len(boxes) == 0:
            print(">>> Result: No pens detected.")
            continue

        print(f">>> Found {len(boxes)} object(s):")
        for box in boxes:
            class_id   = int(box.cls.item())
            confidence = float(box.conf.item())
            class_name = model.names[class_id]

            # Explicitly flag uncapped pens as warnings
            status = "WARNING: UNCAPPED" if class_id == 0 else "CAPPED"
            print(f"    - {status} (Confidence: {confidence * 100:.2f}%)")

        if show_image:
            result.show()   # Opens annotated image window


if __name__ == "__main__":
    parser = argparse.ArgumentParser(
        description="Test Pen Cap Detector on an image"
    )
    parser.add_argument('--image', type=str, required=False)
    parser.add_argument('--no-show', action='store_true')

    args = parser.parse_args()
    img_to_test = args.image

    if not img_to_test:
        img_to_test = input("Enter image path: ").strip()\
            .replace('"', '').replace("'", "")

    if os.path.exists(img_to_test):
        detect_pen_cap(img_to_test, show_image=not args.no_show)
    else:
        print(f"Error: '{img_to_test}' not found.")

RELIABILITY FEATURES

Ensuring Real-World Accuracy

Three techniques prevent the model from memorizing training data and ensure it generalizes to unseen factory conditions.

🎨

Visual Augmentation

Training images are dynamically transformed — random flips, brightness shifts, color jitter, and cropping. This forces the model to recognize pens in varied lighting and orientations, mimicking a real factory line camera.

augment=True in model.train()
⏱️

Early Stopping

If validation metrics stop improving for 25 consecutive epochs (patience=25), training halts automatically. This prevents over-fitting — where a model scores perfectly on training data but fails on new images.

patience=25 in model.train()
🎯

Label Smoothing

Instead of hard 0/1 targets, the model trains against softened targets (e.g. 0.05 / 0.95). This prevents over-confidence and helps the model report calibrated confidence scores — a more reliable industrial signal.

label_smoothing=0.1 in model.train()
STAGE 05

Classification & Inference

The trained model is deployed via classify.py. A low confidence threshold minimizes the chance of missing an uncapped pen.

CONFIDENCE THRESHOLD

In production, a missed uncapped pen (false negative) causes defects. A false alarm (false positive) only costs a re-check. So the threshold is intentionally low.

0.15
conf threshold
conf=0.15
0.0 — Detect everything 0.5 — Standard 1.0 — Certain only
📁

Load Trained Weights

The best checkpoint saved during training at runs/detect/pen_cap_model2/weights/best.pt is loaded. "Best" is selected by YOLO based on highest validation mAP.

🖼️

Input Image

Accepts any image path via CLI (--image path/to/img.jpg) or interactive prompt. The image is internally resized to 640×640 for inference.

🔍

Run Detection

YOLO scans the image with conf=0.15 and iou=0.45. All detected pens — along with their class IDs and confidence scores — are returned as bounding box objects.

⚠️

Flag Anomalies

If class_id == 0 (uncapped), output is tagged as WARNING: UNCAPPED. If class_id == 1, logged as CAPPED. The annotated image can be displayed with bounding boxes.
